create a heroku account
https://api.heroku.com/signup

install heroku gem
gem install heroku

heroku create --stack bamboo-mri-1.9.2 --remote staging nimbleshop-staging

git remote

git push staging staging:master

heroku ps --remote staging

heroku rake db:migrate --remote staging

heroku rake db:data:load --remote staging

heroku rake db:seed --remote staging

heroku open --remote staging

heroku config:add RACK_ENV=staging --remote staging

heroku config --remote staging

heroku info --remote staging

heroku config:add LOG_LEVEL=DEBUG

heroku logs --tail --remote staging
